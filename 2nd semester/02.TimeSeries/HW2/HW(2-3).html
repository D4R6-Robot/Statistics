# Holt-Winters add & multiple
# https://otexts.com/fppkr/holt-winters.html

# The time series should have frequency greater than 1. Therefore frequency = 2
books_paperback <- window(ts(books[,c("Paperback")], start = 1, frequency = 2))
paperback_hw_add <- hw(books_paperback, seasonal="additive")
paperback_hw_mul <- hw(books_paperback, seasonal="multiplicative")

p1 <-  autoplot(books_paperback) +
    autolayer(bp_fit_add, series = "hw add predict", PI = FALSE) + 
    autolayer(bp_fit_mul, series = "hw multi predict", PI = FALSE) +
    xlab("frequency") + ylab("Paperback")

books_hardcover <- window(ts(books[,c("Hardcover")], start = 1, frequency = 2))
hardcover_hw_add <- hw(books_hardcover, seasonal="additive")
hardcover_hw_mul <- hw(books_hardcover, seasonal="multiplicative")

p2 <- autoplot(books_hardcover) +
    autolayer(bh_fit_add, series = "hw add predict", PI = FALSE) + 
    autolayer(bh_fit_mul, series = "hw multi predict", PI = FALSE) +
    xlab("frequency") + ylab("Hardcover")

grid.arrange(p1, p2)
